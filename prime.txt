function checkPrime(num){
let count=0;
     for(let i=1; i<=num; i++){
         if (num%i==0){
             count++;
           }
if(count==1){
return true;
       }
else {return false;}
}
}
let ans =checkPrime(14)
if(ans==true){console.log("prime");}
else {console.log("not prime")
}
